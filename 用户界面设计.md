# 用户界面设计

## 1. 系统概述

用户界面是macOS本地化智能体编码工具的核心组件之一，负责提供直观、高效、美观的用户交互体验，支持macOS原生设计语言，确保与系统的一致性和融合度。

## 2. 设计原则

### 2.1 macOS设计语言

- **清晰**：简洁明了的界面布局，减少视觉干扰
- **遵从**：遵循macOS人机界面指南
- **深度**：通过阴影和分层创造视觉深度
- **响应**：即时反馈和流畅的动画效果
- **一致性**：保持与macOS系统应用的视觉一致性

### 2.2 可用性原则

- **直观**：界面元素易于理解和使用
- **高效**：减少操作步骤，提高工作效率
- **容错**：提供明确的错误提示和恢复机制
- **可访问**：支持辅助功能，确保所有用户可访问
- **个性化**：支持用户自定义界面布局和设置

### 10.1 主应用初始化

```typescript
// main.ts
import { app, BrowserWindow } from 'electron';
import * as path from 'path';
import * as url from 'url';
import { Application } from './application';

let mainWindow: BrowserWindow | null = null;
let application: Application | null = null;

function createWindow() {
  // 创建主窗口
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    title: 'Smart Agent Coder',
    backgroundColor: '#ffffff',
    frame: true,
    show: false,
    webPreferences: {
      nodeIntegration: true,
      contextIsolation: false,
      enableRemoteModule: true,
      devTools: process.env.NODE_ENV === 'development'
    }
  });

  // 加载应用
  if (process.env.NODE_ENV === 'development') {
    mainWindow.loadURL('http://localhost:3000');
    mainWindow.webContents.openDevTools();
  } else {
    mainWindow.loadURL(url.format({
      pathname: path.join(__dirname, '../renderer/index.html'),
      protocol: 'file:',
      slashes: true
    }));
  }

  // 窗口准备就绪后显示
  mainWindow.once('ready-to-show', () => {
    if (mainWindow) {
      mainWindow.show();
    }
  });

  // 窗口关闭事件
  mainWindow.on('closed', () => {
    mainWindow = null;
  });

  // 初始化应用
  application = new Application(mainWindow);
  application.start();
}

// 应用就绪事件
app.on('ready', createWindow);

// 所有窗口关闭事件
app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

// 应用激活事件
app.on('activate', () => {
  if (mainWindow === null) {
    createWindow();
  }
});
```

interface ThemeContextType {
  theme: Theme;
  themes: Theme[];
  setTheme: (id: string) => void;
  createTheme: (name: string, base: string, overrides: Partial<Theme>) => void;
  deleteTheme: (id: string) => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

// 主题提供者组件
export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [themes, setThemes] = useState<Theme[]>(defaultThemes);
  const [currentThemeId, setCurrentThemeId] = useState<string>('auto');
  const [systemTheme, setSystemTheme] = useState<'light' | 'dark'>('light');

  // 监听系统主题变化
  useEffect(() => {
    if (process.platform === 'darwin') {
      const { systemPreferences } = require('electron');
      setSystemTheme(systemPreferences.isDarkMode() ? 'dark' : 'light');

      const listener = () => {
        setSystemTheme(systemPreferences.isDarkMode() ? 'dark' : 'light');
      };

      systemPreferences.on('color-scheme-changed', listener);
      return () => systemPreferences.off('color-scheme-changed', listener);
    }
  }, []);

  // 获取当前主题
  const getCurrentTheme = (): Theme => {
    if (currentThemeId === 'auto') {
      const themeId = systemTheme === 'dark' ? 'dark' : 'light';
      return themes.find(t => t.id === themeId) || themes[0];
    }
    return themes.find(t => t.id === currentThemeId) || themes[0];
  };

  // 设置主题
  const setTheme = (id: string) => {
    setCurrentThemeId(id);
    // 保存主题设置
    localStorage.setItem('theme', id);
  };

  // 创建主题
  const createTheme = (name: string, base: string, overrides: Partial<Theme>) => {
    const baseTheme = themes.find(t => t.id === base);
    if (baseTheme) {
      const newTheme: Theme = {
        id: `custom-${Date.now()}`,
        name,
        type: baseTheme.type,
        colors: { ...baseTheme.colors, ...overrides.colors },
        fonts: { ...baseTheme.fonts, ...overrides.fonts },
        components: { ...baseTheme.components, ...overrides.components }
      };
      setThemes([...themes, newTheme]);
    }
  };

  // 删除主题
  const deleteTheme = (id: string) => {
    if (id !== 'light' && id !== 'dark' && id !== 'auto') {
      setThemes(themes.filter(t => t.id !== id));
      if (currentThemeId === id) {
        setCurrentThemeId('auto');
      }
    }
  };

  // 加载保存的主题
  useEffect(() => {
    const savedTheme = localStorage.getItem('theme');
    if (savedTheme) {
      setCurrentThemeId(savedTheme);
    }
  }, []);

  const theme = getCurrentTheme();

  return (
    <ThemeContext.Provider value={{ theme, themes, setTheme, createTheme, deleteTheme }}>
      {children}
    </ThemeContext.Provider>
  );
};

// 主题钩子
export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (context === undefined) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
};

const AgentChat: React.FC = () => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState('');

      sender: 'user',
      timestamp: new Date()

// 终端组件
const Terminal: React.FC = () => {
  const [commands, setCommands] = useState<Command[]>([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  const executeCommand = async () => {
    if (!input.trim()) return;

    // 添加命令到历史
    const command: Command = {
      id: Date.now().toString(),
      text: input,
      output: '',
      timestamp: new Date(),
      status: 'running'
    };

    setCommands([...commands, command]);
    setInput('');
    setIsLoading(true);

    try {
      // 执行命令
      const output = await window.electron.ipcRenderer.invoke('console:executeCommand', input);

      // 更新命令状态
      setCommands(commands.map(cmd => 
        cmd.id === command.id 
          ? { ...cmd, output, status: 'completed' }
          : cmd
      ));
    } catch (error) {
      // 更新命令状态为错误
      setCommands(commands.map(cmd => 
        cmd.id === command.id 
          ? { ...cmd, output: (error as Error).message, status: 'error' }
          : cmd
      ));
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="terminal">
      {/* 命令历史 */}
      <div className="command-history">
        {commands.map(command => (
          <div key={command.id} className="command">
            <div className="command-input">$ {command.text}</div>
            <div className={`command-output ${command.status}`}>
              {command.output}
            </div>
          </div>
        ))}
        {isLoading && <div className="loading">执行中...</div>}
      </div>

      {/* 输入区域 */}
      <div className="terminal-input">
        <span className="prompt">$</span>
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === 'Enter' && executeCommand()}
          autoFocus
        />
      </div>
    </div>
  );
};
