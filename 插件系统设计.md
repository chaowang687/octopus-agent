# 插件系统设计

## 1. 系统概述

插件系统是macOS本地化智能体编码工具的核心组件之一，负责管理和扩展应用功能，支持插件的下载、安装、卸载、更新和配置，为用户提供灵活的功能扩展能力。

## 2. 功能需求

### 2.1 核心功能
- 插件仓库管理
- 插件安装与卸载
- 插件更新与版本管理
- 插件配置与设置
- 插件依赖解析

### 2.2 高级功能
- 插件市场浏览与搜索
- 插件评分与评论
- 插件开发与发布工具
- 插件安全验证
- 插件性能监控

### 3.2 核心组件

#### 3.2.1 插件仓库客户端
- **功能**：与插件仓库交互，获取插件信息和下载插件
- **实现方案**：
  - 使用HTTP/HTTPS协议与插件仓库通信
  - 支持多种插件仓库源
  - 实现插件搜索和过滤
  - 提供插件详情和版本信息

- **功能**：负责插件的安装和卸载

  - 支持多种插件格式（zip、tar.gz等）
  - 实现插件解压和部署
  - 处理插件安装依赖
  - 提供安装进度和状态

- **功能**：管理已安装的插件，包括启用、禁用、配置等

  - 管理插件生命周期
  - 处理插件配置

- **功能**：解析和管理插件依赖关系

## 6. 技术实现方案

### 6.1 插件安装流程

```typescript
// PluginInstaller.ts
import * as fs from 'fs-extra';
import * as path from 'path';
import AdmZip from 'adm-zip';
import semver from 'semver';
import { Plugin, PluginManifest } from './types';

export class PluginInstaller {
  private installDir: string;
  private progress: number = 0;

  constructor(installDir: string = path.join(process.env.HOME || '', '.smart-agent-coder', 'plugins')) {
    this.installDir = installDir;
    fs.ensureDirSync(this.installDir);
  }

  async install(pluginPath: string, options: InstallOptions = {}): Promise<Plugin> {
    try {
      this.progress = 0;
      
      // 验证插件包
      this.progress = 10;
      const isValid = await this.validatePluginPackage(pluginPath);
      if (!isValid) {
        throw new Error('Invalid plugin package');
      }
      
      // 解压插件包
      this.progress = 30;
      const extractDir = path.join(this.installDir, 'temp');
      await this.extractPluginPackage(pluginPath, extractDir);
      
      // 读取插件清单
      this.progress = 50;
      const manifestPath = path.join(extractDir, 'plugin.json');
      if (!fs.existsSync(manifestPath)) {
        throw new Error('Missing plugin.json manifest file');
      }
      
      const manifest: PluginManifest = JSON.parse(await fs.readFile(manifestPath, 'utf8'));
      const pluginName = manifest.name;
      const pluginVersion = manifest.version;
      
      // 确定安装目录
      const pluginDir = path.join(this.installDir, pluginName);
      
      // 如果插件已存在且未强制安装，则抛出错误
      if (fs.existsSync(pluginDir) && !options.force) {
        throw new Error(`Plugin ${pluginName} already installed`);
      }
      
      // 清理旧版本
      if (fs.existsSync(pluginDir)) {
        await fs.remove(pluginDir);
      }
      
      // 移动插件文件
      this.progress = 70;
      await fs.move(extractDir, pluginDir);
      
      // 安装依赖
      if (!options.skipDependencies) {
        this.progress = 80;
        await this.installDependencies({
          name: pluginName,
          version: pluginVersion,
          description: manifest.description,
          author: manifest.author,
          path: pluginDir,
          enabled: true,
          config: {},
          dependencies: manifest.dependencies || {},
          main: manifest.main,
          entryPoint: manifest.entryPoint || manifest.main,
          manifest
        });
      }
      
      this.progress = 100;
      
      // 返回插件对象
      return {
        name: pluginName,
        version: pluginVersion,
        description: manifest.description,
        author: manifest.author,
        path: pluginDir,
        enabled: true,
        config: {},
        dependencies: manifest.dependencies || {},
        main: manifest.main,
        entryPoint: manifest.entryPoint || manifest.main,
        manifest
      };
    } catch (error) {
      // 清理临时文件
      const tempDir = path.join(this.installDir, 'temp');
      if (fs.existsSync(tempDir)) {
        await fs.remove(tempDir);
      }
      throw error;
    }
  }

  async uninstall(name: string): Promise<void> {
    const pluginDir = path.join(this.installDir, name);
    if (fs.existsSync(pluginDir)) {
      await fs.remove(pluginDir);
    } else {
      throw new Error(`Plugin ${name} not found`);
    }
  }

  async validatePluginPackage(pluginPath: string): Promise<boolean> {
    // 检查文件是否存在
    if (!fs.existsSync(pluginPath)) {
      return false;
    }
    
    // 检查文件扩展名
    const ext = path.extname(pluginPath);
    if (!['.zip', '.tar', '.tar.gz'].includes(ext)) {
      return false;
    }
    
    return true;
  }

  async extractPluginPackage(pluginPath: string, targetDir: string): Promise<void> {
    // 确保目标目录存在
    await fs.ensureDir(targetDir);
    
    // 根据文件类型解压
    const ext = path.extname(pluginPath);
    if (ext === '.zip') {
      const zip = new AdmZip(pluginPath);
      zip.extractAllTo(targetDir, true);
    } else {
      // 处理tar和tar.gz文件
      // 实现略...
    }
  }

  async installDependencies(plugin: Plugin): Promise<void> {
    const packageJsonPath = path.join(plugin.path, 'package.json');
    if (fs.existsSync(packageJsonPath)) {
      // 使用npm安装依赖
      const { exec } = require('child_process');
      const { promisify } = require('util');
      const execAsync = promisify(exec);
      
      await execAsync('npm install', { cwd: plugin.path });
    }
  }

  getInstallationProgress(): number {
    return this.progress;
  }
}

interface InstallOptions {
  force?: boolean;
  skipDependencies?: boolean;
  silent?: boolean;
}
```

### 6.2 插件加载与启动

```typescript
// PluginManager.ts
import * as fs from 'fs-extra';
import * as path from 'path';
import { Plugin, PluginStatus } from './types';

export class PluginManager {
  private plugins: Map<string, Plugin> = new Map();
  private pluginStatuses: Map<string, PluginStatus> = new Map();
  private installDir: string;

  constructor(installDir: string = path.join(process.env.HOME || '', '.smart-agent-coder', 'plugins')) {
    this.installDir = installDir;
    fs.ensureDirSync(this.installDir);
  }

  async loadPlugins(): Promise<void> {
    // 读取插件目录
    const pluginDirs = await fs.readdir(this.installDir);
    
    for (const dir of pluginDirs) {
      const pluginPath = path.join(this.installDir, dir);
      
      // 检查是否为目录
      if (!fs.statSync(pluginPath).isDirectory()) {
        continue;
      }
      
      // 检查是否有插件清单文件
      const manifestPath = path.join(pluginPath, 'plugin.json');
      if (!fs.existsSync(manifestPath)) {
        continue;
      }
      
      try {
        // 读取插件清单
        const manifest = JSON.parse(await fs.readFile(manifestPath, 'utf8'));
        
        // 创建插件对象
        const plugin: Plugin = {
          name: manifest.name,
          version: manifest.version,
          description: manifest.description,
          author: manifest.author,
          path: pluginPath,
          enabled: true,
          config: {},
          dependencies: manifest.dependencies || {},
          main: manifest.main,
          entryPoint: manifest.entryPoint || manifest.main,
          manifest
        };
        
        // 加载插件配置
        const configPath = path.join(pluginPath, 'config.json');
        if (fs.existsSync(configPath)) {
          plugin.config = JSON.parse(await fs.readFile(configPath, 'utf8'));
        }
        
        // 注册插件
        this.registerPlugin(plugin);
        
        // 启动已启用的插件
        if (plugin.enabled) {
          await this.startPlugin(plugin.name);
        }
      } catch (error) {
        console.error(`Error loading plugin ${dir}:`, error);
      }
    }
  }

  registerPlugin(plugin: Plugin): void {
    this.plugins.set(plugin.name, plugin);
    this.pluginStatuses.set(plugin.name, {
      name: plugin.name,
      status: 'idle',
      uptime: 0,
      memory: 0,
      cpu: 0
    });
  }

  unregisterPlugin(name: string): void {
    this.plugins.delete(name);
    this.pluginStatuses.delete(name);
  }

  getPlugin(name: string): Plugin | null {
    return this.plugins.get(name) || null;
  }

  getPlugins(): Plugin[] {
    return Array.from(this.plugins.values());
  }

  getEnabledPlugins(): Plugin[] {
    return Array.from(this.plugins.values()).filter(plugin => plugin.enabled);
  }

  async enablePlugin(name: string): Promise<void> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    plugin.enabled = true;
    await this.startPlugin(name);
  }

  async disablePlugin(name: string): Promise<void> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    await this.stopPlugin(name);
    plugin.enabled = false;
  }

  async startPlugin(name: string): Promise<void> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    try {
      // 更新状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'starting'
      });
      
      // 加载插件入口文件
      const entryPointPath = path.join(plugin.path, plugin.entryPoint);
      if (!fs.existsSync(entryPointPath)) {
        throw new Error(`Entry point ${plugin.entryPoint} not found`);
      }
      
      // 动态加载插件
      const PluginClass = require(entryPointPath);
      const pluginInstance = new PluginClass.default({
        // 插件上下文
        commands: {
          registerCommand: (commandId, handler) => {
            // 注册命令实现
          }
        },
        window: {
          showInformationMessage: (message) => {
            console.log(`[${plugin.name}] ${message}`);
          }
        },
        events: {
          on: (event, handler) => {
            // 事件监听实现
          }
        },
        config: plugin.config
      });
      
      // 激活插件
      if (pluginInstance.activate) {
        await pluginInstance.activate();
      }
      
      // 更新状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'running',
        uptime: Date.now()
      });
    } catch (error) {
      // 更新错误状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'error',
        error: (error as Error).message
      });
      throw error;
    }
  }

  async stopPlugin(name: string): Promise<void> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    try {
      // 更新状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'stopping'
      });
      
      // 加载插件入口文件
      const entryPointPath = path.join(plugin.path, plugin.entryPoint);
      if (fs.existsSync(entryPointPath)) {
        const PluginClass = require(entryPointPath);
        const pluginInstance = new PluginClass.default({});
        
        // 停用插件
        if (pluginInstance.deactivate) {
          await pluginInstance.deactivate();
        }
      }
      
      // 更新状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'idle',
        uptime: 0
      });
    } catch (error) {
      console.error(`Error stopping plugin ${name}:`, error);
      // 更新错误状态
      this.pluginStatuses.set(name, {
        ...this.pluginStatuses.get(name)!, 
        status: 'error',
        error: (error as Error).message
      });
    }
  }

  async restartPlugin(name: string): Promise<void> {
    await this.stopPlugin(name);
    await this.startPlugin(name);
  }

  async configurePlugin(name: string, config: Record<string, any>): Promise<void> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    // 更新配置
    plugin.config = { ...plugin.config, ...config };
    
    // 保存配置
    const configPath = path.join(plugin.path, 'config.json');
    await fs.writeFile(configPath, JSON.stringify(plugin.config, null, 2));
    
    // 重启插件以应用配置
    if (plugin.enabled) {
      await this.restartPlugin(name);
    }
  }

  async getPluginConfig(name: string): Promise<Record<string, any>> {
    const plugin = this.plugins.get(name);
    if (!plugin) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    return plugin.config;
  }

  async checkPluginStatus(name: string): Promise<PluginStatus> {
    const status = this.pluginStatuses.get(name);
    if (!status) {
      throw new Error(`Plugin ${name} not found`);
    }
    
    // 更新资源使用情况
    if (status.status === 'running') {
      // 实现资源使用监控
      status.memory = process.memoryUsage().heapUsed / 1024 / 1024;
      status.cpu = 0; // 简化实现
    }
    
    return status;
  }
}
```
