      console.error('[SelfCorrection] Failed to load error patterns:', e)
      this.errorPatterns = []
    }
  }
 
  /**
   * 设置最大重试次数
   */
  setMaxRetries(max: number): void {
    this.maxRetries = Math.max(1, Math.min(max, 10))
  }
