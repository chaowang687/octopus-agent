    const { unifiedReasoningEngine } = await import('./UnifiedReasoningEngine')
    
    if (unifiedReasoningEngine) {
      // 评估任务复杂度

      // 推荐推理模式

        enableDeepReflection: complexity !== 'low',
        enableToolSelection: true,

        enableTreeReasoning: complexity === 'high'

      // 执行推理
      const reasoningResult = await unifiedReasoningEngine.reason(instruction, reasoningOptions)
      
      // 根据推理结果执行任务
      if (reasoningResult.success) {
        // 使用推理结果
        return {
          success: true,
          result: { response: { message: reasoningResult.answer } },
          reasoningMode: reasoningResult.mode
        }
      }
    }
  } catch (error) {

```typescript
// 集成点：高复杂度任务处理
private async executeComplexTask(instruction: string, model: string = 'openai', agentOptions?: AgentOptions): Promise<any> {
  try {
    const { thoughtTreeEngine } = await import('./ThoughtTreeEngine')
    
    if (thoughtTreeEngine) {
      const totOptions = {
        maxIterations: 15,
        temperature: 0.7,
        maxDepth: 5,
        branchStrategy: {
          type: 'adaptive',
          beamWidth: 3,
          maxBranches: 5,
          confidenceThreshold: 0.3,
          diversityBonus: 0.2
        },
        enableBacktracking: true,
        enableParallelExploration: true,
        maxParallelBranches: 3
      }
      
      const tree = await thoughtTreeEngine.execute(instruction, totOptions, (node, tree) => {
        // 发送思维树节点到前端
        this.emit('progress', {
          taskId,
          type: 'thought_tree_node',
          timestamp: Date.now(),
          node: {
            id: node.id,
            parentId: node.parentId,
            depth: node.depth,
            thought: node.thought,
            action: node.action,
            confidence: node.confidence,
            status: node.status,
            timestamp: node.timestamp
          },
          treeId: tree.id
        } satisfies TaskProgressEvent)
      })
      
      const bestPath = tree.bestPath
      const finalAnswer = bestPath[bestPath.length - 1]?.thought || 'Task incomplete'
      
      return {
        success: true,
        result: { response: { message: finalAnswer } },
        thoughtTree: tree
      }
    }
  } catch (error) {
    console.error('ThoughtTreeEngine集成失败:', error)
    // 回退到EnhancedReActEngine
  }
  
  // 回退到EnhancedReActEngine
  return this.executeEnhancedReActTask(instruction, model, agentOptions)
}
```
