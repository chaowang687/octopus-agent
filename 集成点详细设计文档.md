```typescript
// 集成点：TaskEngine.executeTask方法
async executeTask(instruction: string, model: string = 'openai', agentOptions?: AgentOptions): Promise<any> {
  let selectedModel: string = model
  const taskId = `task_${Date.now()}`
  let routingDecision: RoutingDecision
  let targetSystem: 'system1' | 'system2'
  let emotion: any = null
  
  try {
    let complexity: 'low' | 'medium' | 'high' = 'medium'
    
    // === 集成：UnifiedReasoningEngine任务复杂度评估 ===
    try {
      const { unifiedReasoningEngine } = await import('./UnifiedReasoningEngine')
      
      if (unifiedReasoningEngine) {
        console.log(`[TaskEngine] 启动任务复杂度评估`)
        
        // 评估任务复杂度
        const complexityAssessment = await unifiedReasoningEngine.assessComplexity(instruction)
        complexity = complexityAssessment.complexity
        
        console.log(`[TaskEngine] 复杂度评估结果: ${complexity}, 置信度: ${complexityAssessment.confidence.toFixed(2)}`)
        console.log(`[TaskEngine] 评估原因: ${complexityAssessment.reasoning}`)
        
        // 发送复杂度评估事件
        this.emit('progress', {
          taskId,
          type: 'complexity_assessment',
          timestamp: Date.now(),
          complexity: complexity,
          confidence: complexityAssessment.confidence,
          reasoning: complexityAssessment.reasoning,
          factors: complexityAssessment.factors
        } satisfies TaskProgressEvent)
      }
    } catch (complexityError: any) {
      console.error('复杂度评估失败:', complexityError)
      // 评估失败时，使用默认复杂度
      console.log(`[TaskEngine] 复杂度评估失败，使用默认复杂度: ${complexity}`)
    }
    
    // 现有的系统路由逻辑
    if (agentOptions?.system) {
      targetSystem = agentOptions.system as 'system1' | 'system2'
      routingDecision = {
        decisionId: `manual_${Date.now()}`,
        selectedSystem: targetSystem,
        confidence: 1.0,
        reason: '用户手动指定系统',
        durationMs: 0
      }
      complexity = (agentOptions?.complexity as 'low' | 'medium' | 'high') || complexity
    } else {
      const emotionDecision = await cognitiveEngine.routeWithEmotion(instruction, {
        dialogueHistory: this.history.map(m => m.role === 'user' ? `[用户]: ${m.content}` : `[助手]: ${m.content}`)
      })
      
      emotion = emotionDecision.emotion
      
      routingDecision = {
        decisionId: emotionDecision.decisionId,
        selectedSystem: emotionDecision.selectedSystem,
        confidence: emotionDecision.confidence,
        reason: `${emotionDecision.reason} | 情绪: ${emotionProcessor.emotionToString(emotionDecision.emotion)}`,
        durationMs: emotionDecision.durationMs
      }
      targetSystem = emotionDecision.selectedSystem
    }
    
    cognitiveEngine.createTrace(instruction, targetSystem)
    
    // 后续代码...
  }
}
```

```typescript
// 集成点：TaskEngine.executeSystem2Task方法
private async executeSystem2Task(
  instruction: string, 
  model: string = 'openai', 
  agentOptions?: AgentOptions,
  routingDecision?: RoutingDecision
): Promise<any> {
  try {
    // 现有代码...
    
    // === 集成：UnifiedReasoningEngine推理执行 ===
    try {
      const { unifiedReasoningEngine, ReasoningMode } = await import('./UnifiedReasoningEngine')
      
      if (unifiedReasoningEngine && complexity !== 'low') {
        console.log(`[TaskEngine] 使用UnifiedReasoningEngine执行推理`)
        
        const reasoningOptions = {
          mode: undefined, // 自动选择
          taskComplexity: complexity,
          enableDeepReflection: complexity !== 'low',
          enableToolSelection: true,
          enableExperienceLearning: true,
          enableSelfConsistency: complexity === 'high',
          enableTreeReasoning: complexity === 'high',
          maxIterations: 15,
          maxDepth: 5,
          temperature: 0.7
        }
        
        // 执行推理
        const reasoningResult = await unifiedReasoningEngine.reason(instruction, reasoningOptions)
        
        // 发送推理结果事件
        this.emit('progress', {
          taskId,
          type: 'reasoning_completed',
          timestamp: Date.now(),
          reasoningMode: reasoningResult.mode,
          success: reasoningResult.success,
          confidence: reasoningResult.confidence,
          statistics: reasoningResult.statistics
        } satisfies TaskProgressEvent)
        
        if (reasoningResult.success && reasoningResult.answer) {
          // 推理成功，返回结果
          return {
            success: true,
            requestedModel: model,
            modelUsed: selectedModel,
            routingDecision: routingDecision ? {
              system: routingDecision.selectedSystem,
              confidence: routingDecision.confidence,
              reason: routingDecision.reason
            } : undefined,
            plan: { 
              steps: [], 
              reasoning: `使用${ReasoningMode[reasoningResult.mode]}推理模式` 
            },
            result: {
              response: {
                message: reasoningResult.answer
              }
            },
            reasoningMode: reasoningResult.mode,
            reasoningConfidence: reasoningResult.confidence
          }
        }
        
        console.log(`[TaskEngine] 推理执行完成，继续执行传统流程`)
      }
    } catch (reasoningError: any) {
      console.error('UnifiedReasoningEngine执行失败:', reasoningError)
      // 推理失败时，回退到传统执行流程
      console.log(`[TaskEngine] 推理执行失败，回退到传统执行流程`)
    }
    
    // 传统执行流程...
  }
}
```

```typescript
// Executor.ts中的工具执行方法
async executeTool(toolName: string, parameters: any, context: any): Promise<any> {
  try {
    // === 集成：EnhancedReActEngine智能工具选择 ===
    try {
      const { enhancedReActEngine } = await import('./EnhancedReActEngine')
      
      if (enhancedReActEngine) {
        console.log(`[Executor] 使用智能工具选择`)
        
        // 智能工具选择
        const toolSelection = await enhancedReActEngine.selectBestTool(
          context.task || '',
          context,
          [toolName], // 可用工具列表
          context.previousSteps || []
        )
        
        console.log(`[Executor] 工具选择结果: ${toolSelection.tool}, 置信度: ${toolSelection.confidence}`)
        
        // 发送工具选择事件
        this.emit('toolSelected', {
          tool: toolSelection.tool,
          confidence: toolSelection.confidence,
          reasoning: toolSelection.reasoning,
          alternatives: toolSelection.alternatives
        })

      console.error('智能工具选择失败:', toolSelectionError)
      // 工具选择失败时，使用原始工具
      console.log(`[Executor] 智能工具选择失败，使用原始工具: ${toolName}`)
    }
    
    // 传统工具执行...
  }
}
```

```typescript
// SelfCorrectionEngine.ts中的纠正方法
async correct(error: string, tool: string, parameters: any, context: any): Promise<CorrectionResult> {
  try {
    // === 集成：EnhancedReActEngine深度反思 ===
    try {
      const { enhancedReActEngine } = await import('./EnhancedReActEngine')
      
      if (enhancedReActEngine) {
        console.log(`[SelfCorrectionEngine] 使用深度反思机制`)
        
        // 创建模拟的ReActStep
        const mockStep = {
          id: `step_${Date.now()}`,
          type: 'act' as const,
          action: tool,
          actionInput: parameters,
          error: error,
          timestamp: Date.now()
        }
        
        // 执行深度反思
        const reflection = await enhancedReActEngine.performDeepReflection(
          mockStep,
          context.plan || {},
          context.previousSteps || []
        )
        
        console.log(`[SelfCorrectionEngine] 深度反思完成，成功率: ${reflection.success}`)
        
        if (reflection.errorAnalysis) {
          console.log(`[SelfCorrectionEngine] 错误分析: ${reflection.errorAnalysis.type}`)
          console.log(`[SelfCorrectionEngine] 根因: ${reflection.errorAnalysis.rootCause}`)
          console.log(`[SelfCorrectionEngine] 建议修复: ${reflection.errorAnalysis.suggestedFix}`)
        }
        
        if (reflection.nextStepSuggestion) {
          console.log(`[SelfCorrectionEngine] 下一步建议: ${reflection.nextStepSuggestion.action}`)
          
          return {
            success: true,
            action: reflection.nextStepSuggestion.action,
            actionInput: parameters,
            strategy: 'deep_reflection',
            explanation: reflection.nextStepSuggestion.reasoning
          }
        }
      }
    } catch (reflectionError: any) {
      console.error('深度反思失败:', reflectionError)
      // 反思失败时，使用传统纠正方法
      console.log(`[SelfCorrectionEngine] 深度反思失败，使用传统纠正方法`)
    }
    
    // 传统纠正方法...
  }
}
```

## 3. 错误处理与回退机制

### 3.1 错误处理层级

| 错误类型 | 处理方式 | 回退机制 |
|----------|----------|----------|
| 复杂度评估失败 | 记录错误，使用默认复杂度 | 使用默认复杂度：medium |
| 推理模式推荐失败 | 记录错误，使用默认模式 | 使用默认模式：ENHANCED_REACT |
| 推理执行失败 | 记录错误，回退到传统执行 | 执行传统的计划生成和执行流程 |
| 智能工具选择失败 | 记录错误，使用原始工具 | 使用原始工具执行 |
| 深度反思失败 | 记录错误，使用传统纠正 | 使用传统的错误纠正方法 |
| 经验学习失败 | 记录错误，继续执行 | 忽略经验学习失败，继续执行 |

1. **高优先级**：TaskEngine集成（复杂度评估和推理执行）
2. **中优先级**：Executor集成（智能工具选择）
