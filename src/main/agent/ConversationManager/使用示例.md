# 对话管理器使用示例

## 一、初始化对话管理器

```typescript
import { ConversationManager } from './Core/ConversationManager'
import { OmniAgent } from './Core/OmniAgent'
import { MemoryService } from './Core/memory/MemoryService'

// 创建记忆服务
const memoryService = new MemoryService({
  shortTermMaxSize: 100,
  shortTermTTL: 3600000, // 1小时
  mediumTermMaxSize: 1000,
  mediumTermTTL: 86400000, // 24小时
  longTermMaxSize: 10000
})

// 创建全能智能体
const omniAgent = new OmniAgent({
  agentType: 'general',
  permissionLevel: 'super_admin',
  enableDeepReasoning: true,
  enableSelfCorrection: true
})

// 创建对话管理器
const conversationManager = new ConversationManager({
  memoryService,
  omniAgent,
  maxHistorySize: 100,
  contextWindowSize: 20,
  enableIntentAnalysis: true,
  enableStateTracking: true,
  enableMemory: true
})
```

## 二、创建会话

```typescript
// 创建新会话
const session = await conversationManager.createSession('user_123', {
  projectId: 'project_456',
  enableMemory: true,
  enableIntentAnalysis: true,
  maxHistorySize: 100,
  contextWindowSize: 20
})

console.log('会话已创建:', session.id)
```

## 三、处理消息

```typescript
// 处理用户消息
```

## 五、事件监听

```typescript
// 监听会话创建事件
conversationManager.on('session_created', (event) => {
  console.log('新会话创建:', event.sessionId)
})

// 监听消息接收事件
conversationManager.on('message_received', (event) => {
  console.log('收到消息:', event.data.message)
})

// 监听意图分析事件
conversationManager.on('intent_analyzed', (event) => {
  console.log('意图分析结果:', event.data.intent)
})

// 监听状态变化事件
conversationManager.on('state_changed', (event) => {
  console.log('状态变化:', event.from, '->', event.to)
})

// 监听任务完成事件
conversationManager.on('task_completed', (event) => {
  console.log('任务完成:', event.sessionId)
})

// 监听错误事件
conversationManager.on('error', (event) => {
  console.error('发生错误:', event.data.error)
})
```

## 六、多轮对话示例

```typescript
async function multiTurnConversation() {
  const session = await conversationManager.createSession('user_123')

  // 第一轮
  const response1 = await conversationManager.processMessage(
    session.id,
    '我想创建一个React应用'
  )
  console.log('回复1:', response1.content)

  // 第二轮（会自动携带上下文）
  const response2 = await conversationManager.processMessage(
    session.id,
    '帮我添加一个登录页面'
  )
  console.log('回复2:', response2.content)

  // 第三轮
  const response3 = await conversationManager.processMessage(
    session.id,
    '现在帮我添加用户注册功能'
  )
  console.log('回复3:', response3.content)

  // 查看对话历史
  const history = conversationManager.getHistory(session.id)
  console.log('总对话轮次:', history.length)
}
```

## 九、错误处理

```typescript
async function withErrorHandling() {
  const session = await conversationManager.createSession('user_123')

  try {
    const response = await conversationManager.processMessage(
      session.id,
      '帮我完成这个任务'
    )

    if (response.metadata.actions) {
      const failedActions = response.metadata.actions.filter(a => !a.success)
      if (failedActions.length > 0) {
        console.warn('部分动作失败:', failedActions)
      }
    }

    return response
  } catch (error) {
    console.error('处理消息时出错:', error)
    
    // 可以选择重试或降级处理
    return {
      sessionId: session.id,
      messageId: '',
      content: '抱歉，处理您的请求时遇到了问题。请稍后再试。',
      role: 'assistant',
      timestamp: Date.now(),
      metadata: {
        intent: 'error',
        taskType: 'text_processing' as any,
        processingTime: 0,
        confidence: 0.5
      }
    }
  }
}
```

## 十一、完整示例

```typescript
import { ConversationManager } from './Core/ConversationManager'
import { OmniAgent } from './Core/OmniAgent'
import { MemoryService } from './Core/memory/MemoryService'

async function main() {
  // 1. 初始化
  const memoryService = new MemoryService()
  const omniAgent = new OmniAgent()
  const conversationManager = new ConversationManager({
    memoryService,
    omniAgent
  })

  // 2. 设置事件监听
  conversationManager.on('message_received', (event) => {
    console.log('📩 收到消息:', event.data.message)
  })

  conversationManager.on('intent_analyzed', (event) => {
    console.log('🎯 意图:', event.data.intent.primaryIntent)
  })

  conversationManager.on('task_completed', (event) => {
    console.log('✅ 任务完成')
  })

  // 3. 创建会话
  const session = await conversationManager.createSession('user_123')

  // 4. 多轮对话
  const messages = [
    '你好，我想创建一个待办事项应用',
    '帮我添加用户认证功能',
    '现在添加数据持久化',
    '最后帮我生成一些测试用例'
  ]

  for (const msg of messages) {
    console.log('\n👤 用户:', msg)
    const response = await conversationManager.processMessage(session.id, msg)
    console.log('🤖 智能体:', response.content)
  }

  // 5. 查看历史
  const history = conversationManager.getHistory(session.id)
  console.log('\n📚 对话历史:', history.length, '轮')

  // 6. 清理
  await conversationManager.shutdown()
}

main().catch(console.error)
```
